/help -- вызов этого списка
/create [название] -- создать предмет
/remove [название] -- удалить предмет
/users -- вывести список пользователей с их id
/make_admin [user_id] -- сделать пользователя администратором
/remove_admin [user_id] -- забрать админку у пользователя
/ban [user_id] -- заблокировать пользователя
/unban [user_id] -- разблокировать пользователя
